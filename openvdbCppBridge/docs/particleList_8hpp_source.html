19class ParticleList {
20protected:
21 struct Particle {
22 openvdb::Vec3R position, velocity;
23 openvdb::Real radius;
24 };
25
26public:
27 using Real = openvdb::Real;
28 using PosType = openvdb::Vec3R; // required by openvdb::tools::PointPartitioner
29 using AttributeType = Real; // required by openvdb::tools::ParticlesToLevelSet
30
31 constexpr static float PARTICLE_RADIUS = 0.1;
32
33 explicit ParticleList(std::vector<openvdb::Vec3R> &particlePositions) {
34 for (auto &p: particlePositions) {
35 add(p);
36 }
37 }
38
39 void add(const openvdb::Vec3R &p, const openvdb::Real &r = PARTICLE_RADIUS,
40 const openvdb::Vec3R &v = openvdb::Vec3R(0, 0, 0)) {
41 Particle particle{};
42 particle.position = p;
43 particle.radius = r;
44 particle.velocity = v;
45 _particles.push_back(particle);
46 }
47
48 // Return the total number of particles in the list.
49 // Always required!
50 size_t size() const {
51 return this->_particles.size();
52 }
53
54 // Get the world-space position of the nth particle.
55 // Required by rasterizeSpheres().
56 void getPos(size_t n, openvdb::Vec3R &xyz) const {
57 assert(n < this->_particles.size());
58 xyz = this->_particles[n].position;
59 }
60
61 // Get the world-space position and radius of the nth particle.
62 // Required by rasterizeSpheres().
63 void getPosRad(size_t n, openvdb::Vec3R &xyz, openvdb::Real &radius) const {
64 assert(n < this->_particles.size());
65 xyz = this->_particles[n].position;
66 radius = this->_particles[n].radius;
67 }
68
69 // Get the world-space position, radius and velocity of the nth particle.
70 // Required by rasterizeTrails().
71 void getPosRadVel(size_t n, openvdb::Vec3R &xyz, openvdb::Real &radius, openvdb::Vec3R &velocity) const {
72 assert(n < this->_particles.size());
73 xyz = this->_particles[n].position;
74 radius = this->_particles[n].radius;
75 velocity = this->_particles[n].velocity;
76 }
77
78 // Get the value of the nth particle's user-defined attribute (of type @c AttributeType).
79 // Required only if attribute transfer is enabled in ParticlesToLevelSet.
80 void getAtt(size_t n, openvdb::Index32 &att) const { att = openvdb::Index32(n); }
81
82private:
83 std::vector<Particle> _particles{};
84};
